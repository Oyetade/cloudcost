--The diff query. Two windows, one join, ordered by the size of the change:

WITH windows AS (
    SELECT
        service_name,
        meter_category,
        resource_type,
        CASE WHEN run_date <  DATE '2024-12-15' THEN 'before' ELSE 'after' END AS win,
        COUNT(*)            AS rows,
        SUM(pre_tax_cost)   AS cost
    FROM cat.raw_cost
    WHERE pool_name IS NULL
      AND run_date >= DATE '2024-09-15'    -- 3 months either side
      AND run_date <  DATE '2025-03-15'
    GROUP BY 1, 2, 3, 4
),
pivoted AS (
    SELECT
        service_name, meter_category, resource_type,
        SUM(rows) FILTER (WHERE win = 'before') AS rows_before,
        SUM(rows) FILTER (WHERE win = 'after')  AS rows_after,
        SUM(cost) FILTER (WHERE win = 'before') AS cost_before,
        SUM(cost) FILTER (WHERE win = 'after')  AS cost_after
    FROM windows
    GROUP BY 1, 2, 3
)
SELECT
    service_name,
    meter_category,
    resource_type,
    COALESCE(rows_before, 0)                          AS rows_before,
    COALESCE(rows_after,  0)                          AS rows_after,
    ROUND(COALESCE(cost_before, 0)::numeric, 2)       AS cost_before,
    ROUND(COALESCE(cost_after,  0)::numeric, 2)       AS cost_after,
    ROUND((COALESCE(cost_after, 0) - COALESCE(cost_before, 0))::numeric, 2) AS cost_delta,
    CASE
        WHEN rows_before IS NULL THEN 'APPEARED'
        WHEN rows_after  IS NULL THEN 'DISAPPEARED'
        ELSE 'both'
    END                                               AS status
FROM pivoted
ORDER BY ABS(COALESCE(cost_after, 0) - COALESCE(cost_before, 0)) DESC
LIMIT 40;

The feature-importance check is the one that's actually outstanding. Your models.py has feature_importance(), and its docstring says it exists as precisely this sanity check: that the model leans on activity and calendar rather than memorising pool identity.
The problem is that walk_forward refits per origin and doesn't retain the model, so you need a fit outside the harness. In run_pipeline.py, after frame_1a is built:


import catpipe.models as M
from catpipe.baselines import spec_for_frame

spec = spec_for_frame("1a")
m = M.QuantileGBM.for_frame(frame_1a)
m.fit(frame_1a[frame_1a["run_date"] < "2026-06-01"], spec)
fi = m.feature_importance()
print(fi.head(15).to_string(index=False))
print(f"\npool_name share of total gain: "
      f"{fi.loc[fi.feature == 'pool_name', 'gain'].sum() / fi.gain.sum():.1%}")
