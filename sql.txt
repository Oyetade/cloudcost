SELECT
    count(*) FILTER (WHERE r.pool_name IS NOT NULL AND j.pool_name IS NOT NULL) AS in_both,
    count(*) FILTER (WHERE r.pool_name IS NOT NULL AND j.pool_name IS NULL)     AS raw_only,
    count(*) FILTER (WHERE r.pool_name IS NULL AND j.pool_name IS NOT NULL)     AS job_only
FROM
    (SELECT DISTINCT subscription_id, batch_account_name, pool_name
     FROM cat.raw_cost
     WHERE run_date = '2026-01-05' AND pool_name IS NOT NULL) r
FULL OUTER JOIN
    (SELECT DISTINCT subscription_id, batch_account_name, pool_name
     FROM cat.job_cost
     WHERE run_date = '2026-01-05' AND pool_name IS NOT NULL) j
  ON  r.subscription_id     = j.subscription_id
  AND r.batch_account_name  = j.batch_account_name
  AND r.pool_name           = j.pool_name;



WITH r AS (
    SELECT run_date, subscription_id, batch_account_name, pool_name,
           sum(pre_tax_cost) AS raw_cost
    FROM cat.raw_cost
    WHERE pool_name IS NOT NULL
    GROUP BY run_date, subscription_id, batch_account_name, pool_name
),
j AS (
    SELECT run_date, subscription_id, batch_account_name, pool_name,
           sum(cost) AS job_cost
    FROM cat.job_cost
    WHERE pool_name IS NOT NULL
    GROUP BY run_date, subscription_id, batch_account_name, pool_name
),
both AS (
    SELECT r.raw_cost, j.job_cost,
           (j.job_cost - r.raw_cost) / NULLIF(r.raw_cost, 0) AS rel_diff
    FROM r
    JOIN j                              -- inner join: shared pool-days only
      ON  r.run_date            = j.run_date
      AND r.subscription_id     = j.subscription_id
      AND r.batch_account_name  = j.batch_account_name
      AND r.pool_name           = j.pool_name
)
SELECT
    count(*)                                              AS shared_pool_days,
    round(avg(rel_diff)::numeric, 4)                      AS mean_rel_diff,
    round((percentile_cont(0.50) WITHIN GROUP (ORDER BY rel_diff))::numeric, 4) AS median_rel_diff,
    round(min(rel_diff)::numeric, 4)                      AS min_rel_diff,
    round(max(rel_diff)::numeric, 4)                      AS max_rel_diff,
    count(*) FILTER (WHERE job_cost > raw_cost)           AS days_job_exceeds_raw,
    count(*) FILTER (WHERE job_cost = 0)                  AS days_job_zero,
    round(sum(job_cost)::numeric, 2)                      AS total_job_cost,
    round(sum(raw_cost)::numeric, 2)                      AS total_raw_cost,
    round((sum(job_cost) / NULLIF(sum(raw_cost), 0))::numeric, 4) AS agg_ratio
FROM both;

WITH r AS (
    SELECT run_date, subscription_id, batch_account_name, pool_name,
           sum(pre_tax_cost) AS raw_cost
    FROM cat.raw_cost
    WHERE pool_name IS NOT NULL
    GROUP BY run_date, subscription_id, batch_account_name, pool_name
),
j AS (
    SELECT run_date, subscription_id, batch_account_name, pool_name,
           sum(cost) AS job_cost
    FROM cat.job_cost
    WHERE pool_name IS NOT NULL
    GROUP BY run_date, subscription_id, batch_account_name, pool_name
),
idle AS (
    SELECT r.run_date, r.subscription_id, r.batch_account_name, r.pool_name,
           r.raw_cost,
           coalesce(j.job_cost, 0) AS job_cost,
           r.raw_cost - coalesce(j.job_cost, 0) AS idle_cost,
           (j.pool_name IS NULL) AS no_job_at_all
    FROM r
    LEFT JOIN j                          -- keep every raw_cost pool-day
      ON  r.run_date            = j.run_date
      AND r.subscription_id     = j.subscription_id
      AND r.batch_account_name  = j.batch_account_name
      AND r.pool_name           = j.pool_name
)
SELECT
    date_trunc('month', run_date)                      AS month,
    count(*)                                           AS pool_days,
    count(*) FILTER (WHERE no_job_at_all)              AS idle_pool_days,
    round(sum(raw_cost)::numeric, 2)                   AS total_cost,
    round(sum(idle_cost)::numeric, 2)                  AS idle_cost,
    round((sum(idle_cost) / NULLIF(sum(raw_cost),0) * 100)::numeric, 1) AS idle_pct

WITH pool_days AS (
    SELECT
        r.run_date,
        r.pool_name,
        SUM(r.cost)                           AS raw_pool_cost,
        COALESCE(SUM(j.job_cost_sum), 0)      AS attributed_cost
    FROM cat.raw_cost r
    LEFT JOIN (
        SELECT run_date, pool_name, SUM(cost) AS job_cost_sum
        FROM cat.job_cost
        GROUP BY run_date, pool_name
    ) j
      ON j.run_date  = r.run_date
     AND j.pool_name = r.pool_name
    WHERE r.pool_name IS NOT NULL
    GROUP BY r.run_date, r.pool_name
),
non_pool AS (
    SELECT run_date, SUM(cost) AS residual_cost
    FROM cat.raw_cost
    WHERE pool_name IS NULL
    GROUP BY run_date
)
SELECT
    DATE_TRUNC('month', d.run_date)                                   AS month,
    ROUND(SUM(d.attributed_cost)::numeric, 2)                         AS attributed_pool_cost,
    ROUND(SUM(d.raw_pool_cost - d.attributed_cost)::numeric, 2)       AS idle_cost,
    ROUND(MAX(n.residual_cost)::numeric, 2)                           AS residual_cost_1b
FROM pool_days d
LEFT JOIN non_pool n ON n.run_date = d.run_date
GROUP BY 1
ORDER BY 1;

--The diff query. Two windows, one join, ordered by the size of the change:

WITH windows AS (
    SELECT
        service_name,
        meter_category,
        resource_type,
        CASE WHEN run_date <  DATE '2024-12-15' THEN 'before' ELSE 'after' END AS win,
        COUNT(*)            AS rows,
        SUM(pre_tax_cost)   AS cost
    FROM cat.raw_cost
    WHERE pool_name IS NULL
      AND run_date >= DATE '2024-09-15'    -- 3 months either side
      AND run_date <  DATE '2025-03-15'
    GROUP BY 1, 2, 3, 4
),
pivoted AS (
    SELECT
        service_name, meter_category, resource_type,
        SUM(rows) FILTER (WHERE win = 'before') AS rows_before,
        SUM(rows) FILTER (WHERE win = 'after')  AS rows_after,
        SUM(cost) FILTER (WHERE win = 'before') AS cost_before,
        SUM(cost) FILTER (WHERE win = 'after')  AS cost_after
    FROM windows
    GROUP BY 1, 2, 3
)
SELECT
    service_name,
    meter_category,
    resource_type,
    COALESCE(rows_before, 0)                          AS rows_before,
    COALESCE(rows_after,  0)                          AS rows_after,
    ROUND(COALESCE(cost_before, 0)::numeric, 2)       AS cost_before,
    ROUND(COALESCE(cost_after,  0)::numeric, 2)       AS cost_after,
    ROUND((COALESCE(cost_after, 0) - COALESCE(cost_before, 0))::numeric, 2) AS cost_delta,
    CASE
        WHEN rows_before IS NULL THEN 'APPEARED'
        WHEN rows_after  IS NULL THEN 'DISAPPEARED'
        ELSE 'both'
    END                                               AS status
FROM pivoted
ORDER BY ABS(COALESCE(cost_after, 0) - COALESCE(cost_before, 0)) DESC
LIMIT 40;

WITH per_write AS (
    SELECT
        run_date, subscription_id, resource_group_name, resource_type,
        meter, batch_account_name, pool_name,
        update_time,
        sum(cost) AS cost_at_write        -- sum in case grain is still finer
    FROM cat.raw_cost
    GROUP BY 1,2,3,4,5,6,7,8
)
SELECT
    run_date, subscription_id, resource_group_name, resource_type,
    meter, batch_account_name, pool_name,
    count(DISTINCT update_time)              AS writes,
    count(DISTINCT round(cost_at_write::numeric, 6)) AS distinct_cost_values,
    min(cost_at_write)                       AS min_cost,
    max(cost_at_write)                       AS max_cost,
    max(cost_at_write) - min(cost_at_write)  AS cost_delta
FROM per_write
GROUP BY 1,2,3,4,5,6,7
HAVING count(DISTINCT update_time) > 1
ORDER BY abs(max(cost_at_write) - min(cost_at_write)) DESC
LIMIT 100;


WITH per_write AS (
    SELECT
        run_date, subscription_id, resource_group_name, resource_type,
        meter, batch_account_name, pool_name,
        update_time,
        sum(cost) AS cost_at_write
    FROM cat.raw_cost
    GROUP BY 1,2,3,4,5,6,7,8
),
changed AS (
    SELECT
        run_date, subscription_id, resource_group_name, resource_type,
        meter, batch_account_name, pool_name,
        max(cost_at_write) - min(cost_at_write) AS cost_delta
    FROM per_write
    GROUP BY 1,2,3,4,5,6,7
    HAVING count(DISTINCT cost_at_write) > 1
)
SELECT
    date_trunc('month', run_date) AS run_month,
    count(*)                       AS rows_rewritten,
    sum(abs(cost_delta))           AS total_abs_cost_moved
FROM changed
GROUP BY 1
ORDER BY 1;
